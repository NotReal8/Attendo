class MatchResult {
  final String name;
  final double confidence;
  final double similarity;

  MatchResult({required this.name, required this.confidence, required this.similarity});

  String get confidencePercent => '${(confidence * 100).toStringAsFixed(1)}%';
  String get similarityPercent => '${(similarity * 100).toStringAsFixed(1)}%';
}

class FaceDetectionResult {
  final List<MatchResult> matches;
  final Rect? faceBox;
  final double? faceEulerY;
  final double? faceEulerX;
  final double? faceWidthRatio;

  FaceDetectionResult({
    required this.matches,
    this.faceBox,
    this.faceEulerY,
    this.faceEulerX,
    this.faceWidthRatio,
  });
}

class FaceService {
  static final FaceService _i = FaceService._();
  factory FaceService() => _i;
  FaceService._();

  FaceDetector?  _detector;
  OrtSession?    _session;
  String?        _inputName;
  bool           _initializing = false;
  bool           _ready        = false;
  bool           _inferencing  = false;

  static const int    _outputSize     = 512;
  static const double _matchThreshold = 0.35;

  // ── Temporal voting buffer ────────────────────────────────
  // Each frame that clears the match threshold casts one vote for that
  // student. Confirmation fires when _votesRequired votes accumulate
  // within a rolling window of _voteWindow frames.
  //
  // At ~1 frame / 650 ms (attendance_screen _frameMs):
  //   _voteWindow = 10 frames ≈ 6.5 s window
  //   _votesRequired = 5 → confirmed after ~3.3 s of consistent matching
  //
  // Tune these two constants to taste.

  static const int _voteWindow    = 10;
  static const int _votesRequired = 5;

  final Map<String, List<double>> _voteBuffer = {};

  /// Push a similarity score for [name] and return true when confirmed.
  /// The caller (attendance_screen) decides what to do on confirmation.
  bool castVote(String name, double similarity) {
    final buf = _voteBuffer.putIfAbsent(name, () => []);
    buf.add(similarity);
    if (buf.length > _voteWindow) buf.removeAt(0);
    final votes = buf.where((s) => s >= _matchThreshold).length;
    appLog('[Vote] "$name": $votes/$_votesRequired '
        '(latest sim=${(similarity * 100).toStringAsFixed(1)}%)');
    return votes >= _votesRequired;
  }

  /// Returns how many above-threshold votes [name] currently has in the buffer.
  /// Used by attendance_screen to drive the verifying badge counter.
  int voteBufferFor(String name) {
    final buf = _voteBuffer[name];
    if (buf == null) return 0;
    return buf.where((s) => s >= _matchThreshold).length;
  }

  /// Exposed so the UI can show "X / votesRequired" without importing the constant.
  static int get votesRequired => _votesRequired;

  /// Reset a single student's buffer after they are confirmed present.
  void clearVotes(String name) {
    _voteBuffer.remove(name);
    appLog('[Vote] cleared "$name"');
  }

  /// Wipe all vote buffers — call on session end / dispose.
  void clearAllVotes() {
    _voteBuffer.clear();
    appLog('[Vote] all buffers cleared');
  }

  // ── Init ──────────────────────────────────────────────────

  Future<void> init() async {
    if (_ready) {
      appLog('[FaceService] init(): already ready — skipping');
      return;
    }
    if (_initializing) {
      appLog('[FaceService] init(): already initializing — waiting...');
      while (_initializing) {
        await Future.delayed(const Duration(milliseconds: 50));
      }
      appLog('[FaceService] init(): wait done, _ready=$_ready');
      return;
    }
    _initializing = true;
    appLog('[FaceService] init(): starting');
    try {
      appLog('[FaceService] Creating MLKit FaceDetector...');
      _detector = FaceDetector(
        options: FaceDetectorOptions(
          performanceMode: FaceDetectorMode.accurate,
          enableLandmarks: false,
          minFaceSize: 0.05,
        ),
      );
      appLog('[FaceService] FaceDetector created ✅');

      appLog('[FaceService] Initializing OrtEnv...');
      OrtEnv.instance.init();
      appLog('[FaceService] OrtEnv initialized ✅');

      appLog('[FaceService] Loading ONNX model: assets/models/w600k_mbf.onnx...');
      final rawAsset   = await rootBundle.load('assets/models/w600k_mbf.onnx');
      final modelBytes = rawAsset.buffer.asUint8List();
      appLog('[FaceService] Model bytes loaded: ${modelBytes.length} bytes');

      final sessionOptions = OrtSessionOptions()
        ..setInterOpNumThreads(2)
        ..setIntraOpNumThreads(2);
      _session = OrtSession.fromBuffer(modelBytes, sessionOptions);
      appLog('[FaceService] OrtSession created ✅');

      final inputNames  = _session!.inputNames;
      final outputNames = _session!.outputNames;
      appLog('[FaceService] ONNX inputNames=$inputNames  outputNames=$outputNames');
      if (inputNames.isEmpty) {
        throw Exception('ONNX model has no input nodes — check model file.');
      }
      _inputName = inputNames.first;
      appLog('[FaceService] Using input node: "$_inputName"');

      _ready = true;
      appLog('[FaceService] Initialization complete ✅  outputSize=$_outputSize');
    } catch (e, st) {
      appLog('[FaceService] FAIL — init threw: $e');
      appLog('[FaceService] Stack: ${st.toString().split('\n').take(4).join(' | ')}');
    } finally {
      _initializing = false;
      appLog('[FaceService] _initializing reset to false');
    }
  }

  bool get tfliteLoaded => _session != null;
  double get matchThreshold => _matchThreshold;

  // ── Face detection ────────────────────────────────────────

  Future<List<Face>> detectFaces(Uint8List jpegBytes) async {
    appLog('[detectFaces] Called with ${jpegBytes.length} bytes');
    await init();

    if (_detector == null) {
      appLog('[detectFaces] FAIL — detector is null after init');
      return [];
    }

    final tempDir  = await getTemporaryDirectory();
    final tempPath = '${tempDir.path}/det_${DateTime.now().millisecondsSinceEpoch}.jpg';
    final tempFile = File(tempPath);
    await tempFile.writeAsBytes(jpegBytes);

    try {
      final faces = await _detector!.processImage(InputImage.fromFilePath(tempPath));
      appLog('[detectFaces] MLKit returned ${faces.length} face(s) ✅');
      for (int i = 0; i < faces.length; i++) {
        final bb = faces[i].boundingBox;
        appLog('[detectFaces] Face[$i]: left=${bb.left.toInt()} top=${bb.top.toInt()} '
            'w=${bb.width.toInt()} h=${bb.height.toInt()}');
      }
      return faces;
    } catch (e, st) {
      appLog('[detectFaces] FAIL — MLKit threw: $e');
      appLog('[detectFaces] Stack: ${st.toString().split('\n').take(3).join(' | ')}');
      return [];
    } finally {
      try { await tempFile.delete(); } catch (_) {}
    }
  }

  // ── Embedding ─────────────────────────────────────────────
  // Padding: 25% on each side for more forehead/chin context.
  // NCHW [1,3,112,112], BGR channel order, (pixel - 127.5) / 128.0

  Future<Float32List?> embed(Uint8List jpegBytes, Face face) async {
    appLog('[embed] Called — ${jpegBytes.length} bytes');

    if (_inferencing) {
      appLog('[embed] BLOCKED — _inferencing=true');
      return null;
    }
    _inferencing = true;

    final session   = _session;
    final inputName = _inputName;
    if (session == null || inputName == null) {
      appLog('[embed] FAIL — session or inputName is null');
      _inferencing = false;
      return null;
    }

    try {
      final rawDecoded = img.decodeImage(jpegBytes);
      if (rawDecoded == null) {
        appLog('[embed] FAIL — decodeImage() returned null');
        return null;
      }

      final decoded = img.bakeOrientation(rawDecoded);
      appLog('[embed] decoded+bakeOrientation: ${decoded.width}x${decoded.height}');

      // 25% padding
      final padX = (face.boundingBox.width  * 0.25).toInt();
      final padY = (face.boundingBox.height * 0.25).toInt();
      final x = (face.boundingBox.left.toInt() - padX).clamp(0, decoded.width  - 1);
      final y = (face.boundingBox.top.toInt()  - padY).clamp(0, decoded.height - 1);
      final w = (face.boundingBox.width.toInt()  + padX * 2).clamp(1, decoded.width  - x);
      final h = (face.boundingBox.height.toInt() + padY * 2).clamp(1, decoded.height - y);
      appLog('[embed] Crop: x=$x y=$y w=$w h=$h (pad=25%)');

      if (w < 20 || h < 20) {
        appLog('[embed] FAIL — crop too small: ${w}x$h');
        return null;
      }

      final cropped = img.copyCrop(decoded, x: x, y: y, width: w, height: h);
      final resized = img.copyResize(cropped,
          width: 112, height: 112, interpolation: img.Interpolation.linear);

      final inputData = Float32List(1 * 3 * 112 * 112);
      for (int py = 0; py < 112; py++) {
        for (int px = 0; px < 112; px++) {
          final p      = resized.getPixel(px, py);
          final offset = py * 112 + px;
          inputData[0 * 112 * 112 + offset] = (p.b.toDouble() - 127.5) / 128.0;
          inputData[1 * 112 * 112 + offset] = (p.g.toDouble() - 127.5) / 128.0;
          inputData[2 * 112 * 112 + offset] = (p.r.toDouble() - 127.5) / 128.0;
        }
      }

      final inputTensor = OrtValueTensor.createTensorWithDataList(
          inputData, [1, 3, 112, 112]);
      final outputs = await session.runAsync(
          OrtRunOptions(), {inputName: inputTensor});
      inputTensor.release();

      final rawOutput = outputs?[0]?.value;
      if (rawOutput == null) {
        appLog('[embed] FAIL — ONNX output null');
        outputs?.forEach((e) => e?.release());
        return null;
      }

      final List<double> rawList;
      if (rawOutput is List) {
        final inner = rawOutput[0];
        if (inner is List) {
          rawList = inner.map<double>((e) => (e as num).toDouble()).toList();
        } else {
          appLog('[embed] FAIL — unexpected inner type: ${inner.runtimeType}');
          outputs?.forEach((e) => e?.release());
          return null;
        }
      } else {
        appLog('[embed] FAIL — unexpected output type: ${rawOutput.runtimeType}');
        outputs?.forEach((e) => e?.release());
        return null;
      }
      outputs?.forEach((e) => e?.release());

      appLog('[embed] raw output length=${rawList.length}');
      if (rawList.length != _outputSize) {
        appLog('[embed] WARNING — expected $_outputSize dims, got ${rawList.length}');
      }

      final emb = _normalize(Float32List.fromList(rawList));
      appLog('[embed] embedding done ✅ size=${emb.length}');
      return emb;
    } finally {
      _inferencing = false;
    }
  }

  // ── Enrollment — one embedding per photo, not averaged ────
  //
  // Returns List<Float32List> so each photo's embedding is kept separately.
  // Side-angle enrollment photos match side-angle recognition frames;
  // averaging destroys that per-pose information.
  //
  // Storage: use packEmbeddings() before writing to DB; sampleCount records
  // how many are packed so unpackEmbeddings() can split them back out.

  Future<List<Float32List>> embeddingListFromPhotos(List<Uint8List> photos) async {
    appLog('[embeddingList] Called with ${photos.length} photo(s)');
    await init();

    if (_session == null) {
      throw Exception('ONNX model failed to load. Check assets/models/w600k_mbf.onnx');
    }

    final embeddings = <Float32List>[];

    for (int i = 0; i < photos.length; i++) {
      appLog('[embeddingList] ── Photo ${i + 1}/${photos.length} ──');

      final rawDecoded = img.decodeImage(photos[i]);
      if (rawDecoded == null) {
        appLog('[embeddingList] Photo ${i + 1}: FAIL — decodeImage() null');
        continue;
      }

      final decoded   = img.bakeOrientation(rawDecoded);
      final jpegBytes = Uint8List.fromList(img.encodeJpg(decoded, quality: 95));

      List<Face> faces;
      try {
        faces = await detectFaces(jpegBytes);
      } catch (e) {
        appLog('[embeddingList] Photo ${i + 1}: FAIL — detectFaces threw: $e');
        continue;
      }

      if (faces.isEmpty) {
        appLog('[embeddingList] Photo ${i + 1}: FAIL — no face detected');
        continue;
      }

      final face = faces.reduce((a, b) =>
          (a.boundingBox.width * a.boundingBox.height) >
                  (b.boundingBox.width * b.boundingBox.height)
              ? a : b);

      final bb = face.boundingBox;
      appLog('[embeddingList] Photo ${i + 1}: face — '
          'left=${bb.left.toInt()} top=${bb.top.toInt()} '
          'w=${bb.width.toInt()} h=${bb.height.toInt()}');

      final emb = await embed(jpegBytes, face);
      if (emb != null) {
        embeddings.add(emb);
        appLog('[embeddingList] Photo ${i + 1}: embed ✅ size=${emb.length}');
      } else {
        appLog('[embeddingList] Photo ${i + 1}: FAIL — embed returned null');
      }
    }

    appLog('[embeddingList] done — ${embeddings.length}/${photos.length} succeeded');
    return embeddings;
  }

  /// Pack a list of embeddings into one Float32List for DB storage.
  /// sampleCount = list.length; each slot is _outputSize floats.
  static Float32List packEmbeddings(List<Float32List> list) {
    if (list.isEmpty) return Float32List(0);
    final size   = list[0].length;
    final packed = Float32List(size * list.length);
    for (int i = 0; i < list.length; i++) {
      packed.setRange(i * size, (i + 1) * size, list[i]);
    }
    return packed;
  }

  /// Unpack a DB blob into individual embeddings.
  ///
  /// FIX: Old students enrolled before the multi-embedding path will have
  /// a blob of exactly _outputSize floats but sampleCount > 1 (it stored
  /// photo count, not embedding count). Guard: if the blob length equals
  /// _outputSize exactly, always treat it as a single embedding regardless
  /// of sampleCount to avoid splitting it into garbage sub-vectors.
  static List<Float32List> unpackEmbeddings(Float32List packed, int count) {
    if (packed.isEmpty) return [];

    // Backward-compat guard: single averaged embedding stored by old code.
    if (packed.length == _outputSize) {
      appLog('[unpack] blob length == _outputSize — treating as single embedding '
          '(ignoring sampleCount=$count)');
      return [packed];
    }

    if (count <= 0) return [packed]; // fallback: treat whole blob as one

    final size = packed.length ~/ count;
    if (size == 0) {
      appLog('[unpack] WARNING — size=0, count=$count, packed.length=${packed.length}');
      return [packed];
    }
    return List.generate(count,
        (i) => Float32List.sublistView(packed, i * size, (i + 1) * size));
  }

  /// Backward-compatible single-embedding path (averaging fallback).
  /// Used by enrollment flows that have not been updated to the list path.
  Future<Float32List?> embeddingFromPhotos(List<Uint8List> photos) async {
    final list = await embeddingListFromPhotos(photos);
    if (list.isEmpty) return null;
    return list.length == 1 ? list[0] : _normalize(_average(list));
  }

  // ── Recognition with bounding box ────────────────────────
  //
  // Returns raw per-student similarities. Temporal voting is intentionally
  // NOT done here — it belongs in the caller (attendance_screen._onFrame)
  // so it has full control over when to confirm and clear buffers.
  //
  // Per-student matching uses max-similarity across all stored embeddings
  // (unpackEmbeddings) so that each enrolled pose can independently match
  // the current frame.

  Future<FaceDetectionResult> matchFaceWithBox(
    Uint8List jpegBytes,
    List<Student> students,
    int imageWidth,
    int imageHeight,
  ) async {
    await init();

    List<Face> faces;
    try {
      faces = await detectFaces(jpegBytes);
    } catch (e) {
      appLog('[Recog] detectFaces error: $e');
      return FaceDetectionResult(matches: []);
    }

    if (faces.isEmpty) return FaceDetectionResult(matches: []);

    final face = faces.reduce((a, b) =>
        (a.boundingBox.width * a.boundingBox.height) >
                (b.boundingBox.width * b.boundingBox.height)
            ? a : b);

    final normBox = Rect.fromLTWH(
      face.boundingBox.left   / imageWidth,
      face.boundingBox.top    / imageHeight,
      face.boundingBox.width  / imageWidth,
      face.boundingBox.height / imageHeight,
    );

    if (students.isEmpty) {
      appLog('[Recog] Face detected, no students enrolled');
      return FaceDetectionResult(matches: [], faceBox: normBox);
    }

    final queryEmb = await embed(jpegBytes, face);
    if (queryEmb == null) {
      appLog('[Recog] Embedding failed');
      return FaceDetectionResult(matches: [], faceBox: normBox);
    }
    appLog('[Recog] Query embedding OK — comparing against ${students.length} student(s)...');

    final results = <MatchResult>[];
    for (final s in students) {
      if (s.embedding.isEmpty) {
        results.add(MatchResult(name: s.name, confidence: 0, similarity: 0));
        continue;
      }

      // Unpack stored embeddings.
      // unpackEmbeddings handles the backward-compat case internally.
      final stored = unpackEmbeddings(s.embedding, s.sampleCount);

      // Max similarity across all stored embeddings for this student.
      double bestSim = 0;
      for (final storedEmb in stored) {
        if (storedEmb.length != queryEmb.length) {
          appLog('[Recog] WARNING — "${s.name}" dim mismatch: '
              'stored=${storedEmb.length} query=${queryEmb.length} — skip slot');
          continue;
        }
        final sim = _cosine(queryEmb, storedEmb);
        if (sim > bestSim) bestSim = sim;
      }

      final conf = ((bestSim - _matchThreshold) / (1.0 - _matchThreshold))
          .clamp(0.0, 1.0);
      appLog('[Recog] "${s.name}": bestSim=${(bestSim * 100).toStringAsFixed(1)}% '
          'conf=${(conf * 100).toStringAsFixed(1)}% (${stored.length} stored emb)');

      results.add(MatchResult(name: s.name, confidence: conf, similarity: bestSim));
    }
    results.sort((a, b) => b.similarity.compareTo(a.similarity));

    final best = results.first;
    if (best.similarity >= _matchThreshold) {
      appLog('[Recog] MATCH: ${best.name} sim=${best.similarityPercent} ✅');
    } else {
      appLog('[Recog] No match. Best: ${best.name} at ${best.similarityPercent}');
    }

    return FaceDetectionResult(
      matches: results,
      faceBox: normBox,
      faceEulerY: face.headEulerAngleY,
      faceEulerX: face.headEulerAngleX,
      faceWidthRatio: face.boundingBox.width / imageWidth,
    );
  }

  Future<List<MatchResult>> matchFace(
      Uint8List jpegBytes, List<Student> students) async {
    final result = await matchFaceWithBox(jpegBytes, students, 1, 1);
    return result.matches;
  }

  // ── Math helpers ──────────────────────────────────────────

  Float32List _normalize(Float32List e) {
    double norm = 0;
    for (final v in e) norm += v * v;
    norm = sqrt(norm);
    if (norm == 0) {
      appLog('[normalize] WARNING — norm is 0, returning as-is');
      return e;
    }
    final out = Float32List(e.length);
    for (int i = 0; i < e.length; i++) out[i] = e[i] / norm;
    return out;
  }

  double _cosine(Float32List a, Float32List b) {
    double dot = 0, na = 0, nb = 0;
    for (int i = 0; i < a.length; i++) {
      dot += a[i] * b[i];
      na  += a[i] * a[i];
      nb  += b[i] * b[i];
    }
    if (na == 0 || nb == 0) return 0;
    return dot / (sqrt(na) * sqrt(nb));
  }

  Float32List _average(List<Float32List> list) {
    final len = list[0].length;
    final avg = Float32List(len);
    for (final e in list) { for (int i = 0; i < len; i++) avg[i] += e[i]; }
    for (int i = 0; i < len; i++) avg[i] /= list.length;
    return avg;
  }

  void dispose() {
    appLog('[FaceService] dispose() called');
    _detector?.close();
    _session?.release();
    OrtEnv.instance.release();
    _ready = false;
    appLog('[FaceService] disposed ✅');
  }
}
