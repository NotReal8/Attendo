class _AttendanceScreenState extends State<AttendanceScreen>
    with WidgetsBindingObserver {

  CameraController? _controller;
  bool _cameraReady = false;
  bool _processing = false;
  bool _saving = false;

  late AttendanceSession _session;
  final Set<String> _presentNames = {};
  String? _lastDetectedName;
  DateTime? _lastDetectionTime;

  // How often to process a frame (milliseconds)
  static const int _frameIntervalMs = 400;

  @override
  void initState() {
    super.initState();
    WidgetsBinding.instance.addObserver(this);
    _session = AttendanceService.newSession();
    _initCamera();
  }

  Future<void> _initCamera() async {
    final status = await Permission.camera.request();
    if (!status.isGranted) {
      if (mounted) {
        ScaffoldMessenger.of(context).showSnackBar(
          const SnackBar(content: Text('Camera permission denied')),
        );
      }
      return;
    }

    final cameras = await availableCameras();
    if (cameras.isEmpty) return;

    // Prefer front camera for attendance
    final camera = cameras.firstWhere(
      (c) => c.lensDirection == CameraLensDirection.front,
      orElse: () => cameras.first,
    );

    _controller = CameraController(
      camera,
      ResolutionPreset.high,
      enableAudio: false,
      imageFormatGroup: ImageFormatGroup.jpeg,
    );

    await _controller!.initialize();
    if (!mounted) return;

    setState(() => _cameraReady = true);

    // Start image stream
    await _controller!.startImageStream(_onCameraFrame);
  }

  void _onCameraFrame(CameraImage image) async {
    if (_processing) return;
    final now = DateTime.now();
    if (_lastDetectionTime != null &&
        now.difference(_lastDetectionTime!).inMilliseconds < _frameIntervalMs) {
      return;
    }

    _processing = true;
    _lastDetectionTime = now;

    try {
      // Convert CameraImage to JPEG bytes
      final bytes = await _cameraImageToJpeg(image);
      if (bytes == null) {
        _processing = false;
        return;
      }

      final rotation = _getRotation();
      final name = await _service.identifyFace(bytes, rotation);

      if (name != null && mounted) {
        setState(() {
          _presentNames.add(name);
          _lastDetectedName = name;
        });

        // Clear the "detected" badge after 2 seconds
        Future.delayed(const Duration(seconds: 2), () {
          if (mounted && _lastDetectedName == name) {
            setState(() => _lastDetectedName = null);
          }
        });
      }
    } catch (_) {
      // Silently ignore per-frame errors
    } finally {
      _processing = false;
    }
  }

  Future<Uint8List?> _cameraImageToJpeg(CameraImage image) async {
    // camera plugin on Android delivers JPEG when imageFormatGroup = jpeg
    // The plane[0] contains the full JPEG bytes
    if (image.planes.isEmpty) return null;
    return Uint8List.fromList(image.planes[0].bytes);
  }

  InputImageRotation _getRotation() {
    final sensorOrientation =
        _controller?.description.sensorOrientation ?? 0;
    switch (sensorOrientation) {
      case 90:
        return InputImageRotation.rotation90deg;
      case 180:
        return InputImageRotation.rotation180deg;
      case 270:
        return InputImageRotation.rotation270deg;
      default:
        return InputImageRotation.rotation0deg;
    }
  }

  Future<void> _finishSession() async {
    setState(() => _saving = true);
    await _controller?.stopImageStream();

    _session = AttendanceSession(
      date: _session.date,
      label: _session.label,

import 'dart:math';
import 'dart:typed_data';
import 'package:flutter/services.dart';

import 'package:image/image.dart' as img;

class FaceEmbeddingService {
  static final FaceEmbeddingService _instance =
      FaceEmbeddingService._internal();
  factory FaceEmbeddingService() => _instance;
  FaceEmbeddingService._internal();

  Interpreter? _interpreter;
  FaceDetector? _detector;

  // ── Initialise ML Kit detector and TFLite interpreter ──────

  Future<void> init() async {
    // ML Kit face detector — high accuracy mode for enrollment
    _detector = FaceDetector(
      options: FaceDetectorOptions(
        performanceMode: FaceDetectorMode.accurate,
        enableLandmarks: false,
        enableContours: false,
        enableClassification: false,
        minFaceSize: 0.10,
      ),
    );

    // Load MobileFaceNet from bundled assets
    _interpreter = await Interpreter.fromAsset(
      'assets/models/mobilefacenet.tflite',
      options: InterpreterOptions()..threads = 4,
    );
  }

  // ── Generate embedding from an asset image (enrollment) ────

  /// Returns null if no face is detected in the image.
  Future<Float32List?> embeddingFromAsset(String assetPath) async {
    // 1. Load raw bytes
    final byteData = await rootBundle.load(assetPath);
    final bytes = byteData.buffer.asUint8List();

    // 2. Decode to img.Image
    final decoded = img.decodeImage(bytes);
    if (decoded == null) return null;

    // 3. Run ML Kit face detection
    final inputImage = InputImage.fromBytes(
      bytes: bytes,
      metadata: InputImageMetadata(
        size: Size(decoded.width.toDouble(), decoded.height.toDouble()),
        rotation: InputImageRotation.rotation0deg,
        format: InputImageFormat.bgra8888,
      ),
    );

    // Use fromFilePath approach via temp file or use byte approach
    // For assets, we'll detect on the raw decoded image dimensions
    final faces = await _detector!.processImage(inputImage);
    if (faces.isEmpty) return null;

    // 4. Crop the largest face
    final face = faces.reduce(
      (a, b) => _faceArea(a) > _faceArea(b) ? a : b,
    );
    final cropped = _cropAndResize(decoded, face.boundingBox);
    if (cropped == null) return null;

    // 5. Run TFLite inference
    return _runInference(cropped);
  }

  // ── Generate embedding from raw camera frame bytes (recognition) ──

  /// [imageBytes] — JPEG/PNG bytes from camera capture
  /// Returns null if no face detected.
  Future<Float32List?> embeddingFromBytes(
      Uint8List imageBytes, InputImageRotation rotation) async {
    final decoded = img.decodeImage(imageBytes);
    if (decoded == null) return null;

    final inputImage = InputImage.fromBytes(
      bytes: imageBytes,
      metadata: InputImageMetadata(
        size: Size(decoded.width.toDouble(), decoded.height.toDouble()),
        rotation: rotation,
        format: InputImageFormat.bgra8888,
      ),
    );

    final faces = await _detector!.processImage(inputImage);
    if (faces.isEmpty) return null;

    final face = faces.reduce(
      (a, b) => _faceArea(a) > _faceArea(b) ? a : b,
    );
    final cropped = _cropAndResize(decoded, face.boundingBox);
    if (cropped == null) return null;

    return _runInference(cropped);
  }

  /// Returns all detected face bounding boxes (for live overlay drawing)
  Future<List<Rect>> detectFaceRects(
      Uint8List imageBytes, InputImageRotation rotation, Size imageSize) async {
    final inputImage = InputImage.fromBytes(
      bytes: imageBytes,
      metadata: InputImageMetadata(
        size: imageSize,
        rotation: rotation,
        format: InputImageFormat.bgra8888,
      ),
    );
    final faces = await _detector!.processImage(inputImage);
    return faces.map((f) => f.boundingBox).toList();
  }

  // ── Cosine similarity ──────────────────────────────────────

  /// Returns a value between -1 and 1. Treat >= 0.75 as a match.
  static double cosineSimilarity(Float32List a, Float32List b) {
    assert(a.length == b.length, 'Embedding length mismatch');
    double dot = 0, normA = 0, normB = 0;
    for (int i = 0; i < a.length; i++) {
      dot += a[i] * b[i];
      normA += a[i] * a[i];
      normB += b[i] * b[i];
    }
    if (normA == 0 || normB == 0) return 0;
    return dot / (sqrt(normA) * sqrt(normB));
  }

  // ── Private helpers ────────────────────────────────────────

  double _faceArea(Face face) {
    final r = face.boundingBox;
    return r.width * r.height;
  }

  /// Crop bounding box from decoded image and resize to 112×112.
  img.Image? _cropAndResize(img.Image src, Rect box) {
    final x = box.left.toInt().clamp(0, src.width - 1);
    final y = box.top.toInt().clamp(0, src.height - 1);
    final w = box.width.toInt().clamp(1, src.width - x);
    final h = box.height.toInt().clamp(1, src.height - y);
    if (w <= 0 || h <= 0) return null;

    final cropped = img.copyCrop(src, x: x, y: y, width: w, height: h);
    return img.copyResize(cropped, width: 112, height: 112);
  }

  /// Normalise pixels to [-1, 1] and run MobileFaceNet inference.
  Float32List _runInference(img.Image face112) {
    // Build input tensor [1, 112, 112, 3]
    final input = List.generate(
      1,
      (_) => List.generate(
        112,
        (y) => List.generate(
          112,
          (x) {
            final pixel = face112.getPixel(x, y);
            return [
              (pixel.r.toDouble() - 127.5) / 127.5,
              (pixel.g.toDouble() - 127.5) / 127.5,
              (pixel.b.toDouble() - 127.5) / 127.5,
            ];
          },
        ),
      ),
    );

    // Output tensor [1, 128]
    final output = List.generate(1, (_) => List.filled(128, 0.0));
    _interpreter!.run(input, output);

    return Float32List.fromList(output[0]);
  }

  void dispose() {
    _detector?.close();
    _interpreter?.close();
  }
}
